#include <stdio.h>
#include <stdlib.h>
#include <windows.h>
#include <time.h>

// x86-64 Kernel
extern float asmsdot(float *a, float *b, int size);

// C Kernel
float sdot(float* a, float* b, int size);

// Initialize a vector with random floats between 0 and 1
float* init_vector(size_t n) {
    float* v = (float*)malloc(n * sizeof(float));
    if (!v) {
        printf("Memory allocation failed!\n");
        exit(1);
    }
    for (size_t i = 0; i < n; i++) {
        v[i] = (float)rand() / RAND_MAX;
    }
    return v;
}

// Measure average execution time of a float-returning kernel
double time_kernel_float(float (*kernel)(float*, float*, int), float* a, float* b, int size, int runs) {
    LARGE_INTEGER start, end, freq;
    QueryPerformanceFrequency(&freq);

    double total_time = 0.0;
    for (int i = 0; i < runs; i++) {
        QueryPerformanceCounter(&start);
        float res = kernel(a, b, size); // store result to prevent optimization
        QueryPerformanceCounter(&end);

        total_time += (double)(end.QuadPart - start.QuadPart) / freq.QuadPart;
    }
    return total_time / runs;
}

int main() {
    srand((unsigned int)time(NULL));

    int runs = 20;
    size_t sizes[] = {1 << 20, 1 << 24, 1 << 28}; // adjust if memory too large

    for (int s = 0; s < 3; s++) {
        size_t n = sizes[s];
        printf("\nVector size: %zu\n", n);

        float* a = init_vector(n);
        float* b = init_vector(n);

        // --- C kernel timing ---
        double t_c = time_kernel_float(sdot, a, b, (int)n, runs);
        float res_c = sdot(a, b, (int)n);
        printf("C kernel average time: %.6f seconds, result: %.6f\n", t_c, res_c);

        // --- ASM kernel timing ---
        double t_asm = time_kernel_float(asmsdot, a, b, (int)n, runs);
        float res_asm = asmsdot(a, b, (int)n);
        printf("ASM kernel average time: %.6f seconds, result: %.6f\n", t_asm, res_asm);

        free(a);
        free(b);
    }

    return 0;
}
